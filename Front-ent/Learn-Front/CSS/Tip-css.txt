- shape-outside: circle(50% at 50% 50%); : cho chữ chạy bo tròn hình tròn
  &__shape{
    width: 15rem;
    height: 15rem;
    background-color: orangered;
    float: left;

    clip-path: circle(50% at 50% 50%);
    transform: translateX(-3rem);
    margin-top: -1rem;

    shape-outside: circle(50% at 50% 50%);
  }
